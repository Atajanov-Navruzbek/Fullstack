function tubSon(n){
        let arr=[]
    for (let i = 2;i < n ;i++){
        let tubmi = true
        for (let j =2;j<i;j++){
            if (i%j===0){
                tubmi = false;
                   break;
            }
        }
        if (tubmi){
            arr.push(i)
        }
    }    
    return arr
};
console.log(tubSon(14))
